#include <windows.h>
#include <tlhelp32.h>
#include <iostream>

int main() {
    HANDLE hSnapshot = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    if (hSnapshot == INVALID_HANDLE_VALUE) {
        std::cerr << "Failed to create a snapshot of processes. Error code: " << GetLastError() << std::endl;
        return 1;
    }

    PROCESSENTRY32 pe32;
    pe32.dwSize = sizeof(PROCESSENTRY32);

    if (!Process32First(hSnapshot, &pe32)) {
        std::cerr << "Failed to create a snapshot of processes. Error code: " << GetLastError() << std::endl;
        CloseHandle(hSnapshot);
        return 1;
    }

    std::cout << "List of running processes:\n";
    std::cout << "---------------------------------------------\n";

    do {
        std::wcout << L"Process Name: " << pe32.szExeFile
                   << L", PID: " << pe32.th32ProcessID
                   << L", PPID: " << pe32.th32ParentProcessID << std::endl;
    } while (Process32Next(hSnapshot, &pe32));

    CloseHandle(hSnapshot);
    return 0;
}
